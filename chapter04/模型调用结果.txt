def count_mondays_in_november():
    import calendar
    from datetime import date
    year, month = 2025, 11
    target_weekday = 0  # Monday is 0 in weekday()
    first_day = date(year, month, 1)
    first_weekday = first_day.weekday()
    _, num_days = calendar.monthrange(year, month)
    days_until_monday = (7 - first_weekday + target_weekday) % 7
    first_monday = 1 + days_until_monday
    count = ((num_days - first_monday) // 7) + 1
    print(count)

def count_wednesdays(year, month):
    from datetime import date
    count = 0
    for day in range(1, 31):  # November has 30 days
        try:
            d = date(year, month, day)
        except ValueError:
            break
        if d.weekday() == 2:  # 2 corresponds to Wednesday (Monday=0, ..., Sunday=6)
            count += 1
    return count

print(count_wednesdays(2025, 11))
